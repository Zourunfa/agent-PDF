相似度计算方法

方法 范围 特点
余弦相似度 (Cosine) [-1, 1] 本项目使用，关注角度不关注大小
欧氏距离 (Euclidean) [0, ∞] 考虑向量大小和方向
点积 (Dot Product) (-∞, ∞) 计算简单，需要归一化
本项目使用：余弦相似度，因为它能更好地捕获语义相似性，不受向量长度影响。

// 文本嵌入实现
import { AlibabaTongyiEmbeddings } from "@langchain/community/embeddings/alibaba_tongyi";

// 初始化嵌入模型
const embeddings = new AlibabaTongyiEmbeddings({
  apiKey: process.env.QWEN_API_KEY,
  modelName: "text-embedding-v2"
});

// 嵌入单个文本
const vector = await embeddings.embedQuery("什么是人工智能？");
console.log(vector.length);  // 1536

// 嵌入多个文本（批量）
const texts = ["AI 是机器智能", "AI 可以为人类服务"];
const vectors = await embeddings.embedDocuments(texts);
console.log(vectors.length);  // 2
// 向量存储实现
export async createVectorStoreFromChunks(
  pdfId: string,
  chunks: Chunk[]
) {
  // 1. 存储 chunks 到 Redis
  await redis.set(`vector:${pdfId}`, chunks);

  // 2. 创建文档
  const documents = chunks.map(chunk =>
    new Document({
      pageContent: chunk.content,
      metadata: chunk.metadata
    })
  );

  // 3. 创建向量存储
  const vectorStore = await MemoryVectorStore
    .fromDocuments(documents, embeddings);

  // 4. 计算并存储 embeddings
  const embeddingVectors = await embeddings
    .embedDocuments(chunks.map(c => c.content));
  await redis.set(
    `vector:${pdfId}:embeddings`,
    embeddingVectors
  );

  return vectorStore;
}
// 相似度搜索实现
export async searchSimilarDocuments(
  pdfId: string,
  query: string,
  k: number = 4
) {
  // 1. 从 Redis 恢复数据
  const chunks = await redis.get(`vector:${pdfId}`);
  const embeddings = await redis
    .get(`vector:${pdfId}:embeddings`);

  // 2. 如果有缓存 embeddings，直接使用
  let vectorStore;
  if (embeddings && embeddings.length === chunks.length) {
    // 避免重新计算！
    vectorStore = new MemoryVectorStore(embeddings);
    await vectorStore.addVectors(
      embeddings,
      chunks.map(c => new Document({
        pageContent: c.content,
        metadata: c.metadata
      }))
    );
  } else {
    // 重新计算
    vectorStore = await createVectorStoreFromChunks(
      pdfId, chunks
    );
  }

  // 3. 相似度搜索
  const results = await vectorStore
    .similaritySearchWithScore(query, k);

  // 4. 过滤低相似度结果
  const threshold = 0.3;
  const filtered = results.filter(([, score]) =>
    score > threshold
  );

  return filtered.map(([doc]) => doc);
}
